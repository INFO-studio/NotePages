- $(C)'=0$
- $(x^\alpha)'=\alpha x^{\alpha-1}$
- $(\sin x)'=\cos x$
- $(\cos x)'=-\sin x$
- $(\tan x)'=\sec^2x$
- $(\cot x)'=-\csc^2x$
- $(\sec x)'=\sec x \tan x$
- $(\csc x)'=-\csc x \cot x$
- $(a^x)'=a^x\ln a$
- $(e^x)'=e^x$
- $(\log_ax)'=\cfrac{1}{x\ln a}$
- $(\ln x)'=\cfrac{1}{x}$
- $(\arcsin x)'=\cfrac{1}{\sqrt{1-x^2}}$
- $(\arctan x)'=\cfrac{1}{1+x^2}$
- $(\arccos x)'=-\cfrac{1}{\sqrt{1-x^2}}$
- $(\mathrm{arccot}\,x)'=-\cfrac{1}{1+x^2}$
